## Usage

```tsx
import { PartialPick } from "@/types/PartialPick";
```

<br />

```ts
type User = {
  id: number;
  email: string;
  name: string;
  created_at: string;
  updated_at: string;
};

type PartialUser = PartialPick<User, "id" | "email">;

// Equivalent to:
type PartialUser = {
  id?: number;
  email?: string;
  name: string;
  created_at: string;
  updated_at: string;
};
```
