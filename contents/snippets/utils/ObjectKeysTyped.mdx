## Usage

```tsx
import { ObjectKeysTyped } from "@/utils/ObjectKeysTyped";
```

<br />

```tsx
const user = {
  name: "Alice",
  age: 30,
  isAdmin: true,
};

const keys = ObjectKeysTyped(user);
console.log(keys); // ["name", "age", "isAdmin"]
```
