## Usage

```tsx
import { useToggleState } from "@/hooks/useToggleState";
```

<br />

```tsx
const [isOpen, setIsOpen, { toggle, activate, deactivate }] =
  useToggleState(false);

// Basic usage
const handleToggle = () => toggle();

// Direct control
const openModal = () => activate();
const closeModal = () => deactivate();
```
