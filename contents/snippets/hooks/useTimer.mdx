## Usage

```tsx
import { useTimeout, useInterval } from "@/registry/hooks/useTimer/useTimer";
```

### timeout

```tsx
const { reset, clear } = useTimeout(
  () => {
    console.log("Timeout finished!");
  },
  5000,
  "interval",
);
```

### interval

```tsx
const { reset, clear } = useInterval(
  () => {
    console.log("Timeout finished!");
  },
  5000,
  "interval",
);
```
