## Usage

```tsx
import { useTimeout, useInterval } from "@/registry/hooks/useTimers/useTimers";
```

### timeout

```tsx
const { reset, clear } = useTimeout(
  () => {
    console.log("Timeout finished!");
  },
  5000,
);
```

### interval

```tsx
const { reset, clear } = useInterval(
  () => {
    console.log("Interval finished!");
  },
  5000,
);
```
