## Usage

```tsx
import useIsVisible from "@/hooks/useIsVisible";
```

<br />

```tsx
const elementRef = useRef<HTMLDivElement>(null);
const isVisible = useIsVisible(elementRef);

// With options
const isVisible = useIsVisible(elementRef, {
  rootMargin: "100px",
  once: true,
});
```
