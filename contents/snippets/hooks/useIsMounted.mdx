## Usage

```tsx
import { useIsMounted } from "@/hooks/useIsMounted";
```

<br />

```tsx
const isMounted = useIsMounted();

// Use to prevent hydration errors
if (!isMounted) {
  return <div>Loading...</div>;
}

return <ClientOnlyComponent />;
```
