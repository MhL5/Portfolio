## Usage

```tsx
import useOnClickOutside from "@/hooks/useOnClickOutside";
```

<br />

```tsx
const modalRef = useRef<HTMLDivElement>(null);

useOnClickOutside(modalRef, (event) => {
  console.log("Clicked outside the modal!");
  setIsOpen(false);
});
```
